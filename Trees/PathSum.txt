/**
 * Definition for binary tree
 * class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) {
 *      val = x;
 *      left=null;
 *      right=null;
 *     }
 * }
 */
public class Solution {
    public int hasPathSum(TreeNode A, int B) {
        if(B==0) return 1;
        if(A==null) return 0;
        if(hasSum(A,B)) return 1;
        return 0;
    }
    
    private boolean hasSum(TreeNode A, int B){
        if(A==null) return false;
        if(A.left==null && A.right==null) return (A.val==B);
        return hasSum(A.left,B-A.val) || hasSum(A.right,B-A.val);
    }
}
